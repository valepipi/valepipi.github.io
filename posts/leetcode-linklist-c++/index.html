class Solution {
public:
    ListNode* partition(ListNode* head, int x) {
        ListNode* left_start = new ListNode(-1);
        ListNode* right_start = new ListNode(-1);
        ListNode* left_end = left_start;
        ListNode* right_end = right_start;
        ListNode* p = head;
        while(p != nullptr){
            if(p->val < x){
                left_end->next = p;
                left_end = left_end->next;
            }else{
                right_end->next = p;
                right_end = right_end->next;
            }
            ListNode* tmp = p->next;
            p->next = nullptr;
            p = tmp;
        }
        left_end->next = right_start->next;
        return left_start->next;
    }
};

            fast = fast->next->next;
            slow = slow->next;

class Solution {
public:
    ListNode* reverseBetween(ListNode* head, int left, int right) {
        if(left == 1) return reverseN(head, right);
        head->next = reverseBetween(head->next, left-1, right-1);
        return head;
    }
    ListNode* succesor = nullptr;
    ListNode* reverseN(ListNode* head, int n){
        if(n == 1){
            succesor = head->next;
            return head;
        }
        ListNode* last = reverseN(head->next, n-1);
        head->next->next = head;
        head->next = succesor;
        return last;
    }
};
