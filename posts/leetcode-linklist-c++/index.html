单链表节点定义

struct ListNode {
    int val;
    ListNode *next;
    ListNode() : val(0), next(nullptr) {}
    ListNode(int x) : val(x), next(nullptr) {}
    ListNode(int x, ListNode *next) : val(x), next(next) {}
};

21.合并两个有序链表

//简单
dummy

class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        ListNode* dummy = new ListNode(-1);
        ListNode* p = dummy;
        ListNode* p1 = list1;
        ListNode* p2 = list2;

        while(p1 != nullptr && p2 != nullptr){
            if(p1->val < p2->val){
                p->next = p1;
                p1 = p1->next;
            }
            else{
                p->next = p2;
                p2 = p2->next;
            }
            p = p->next;
        }

        if(p1 != nullptr){
            p->next = p1;
        }
        if(p2 != nullptr)
        {
            p->next = p2;
        }
        return dummy->next;
    }
};

86.分隔链表

//中等
双指针

class Solution {
public:
    ListNode* partition(ListNode* head, int x) {
        ListNode* left_start = new ListNode(-1);
        ListNode* right_start = new ListNode(-1);
        ListNode* left_end = left_start;
        ListNode* right_end = right_start;
        ListNode* p = head;
        while(p != nullptr){
            if(p->val < x){
                left_end->next = p;
                left_end = left_end->next;
            }else{
                right_end->next = p;
                right_end = right_end->next;
            }
            ListNode* tmp = p->next;
            p->next = nullptr;
            p = tmp;
        }
        left_end->next = right_start->next;
        return left_start->next;
    }
};

19. 删除链表的倒数第 N 个结点

// 快慢指针
dummy node: 避免空指针
比如：一共只有5个节点，要删除倒数第5个节点(n=5)，先找倒数第6个节点，而它不存在

class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        ListNode* dummy = new ListNode(-1);
        dummy->next = head;
        ListNode* p1 = dummy;
        ListNode* p2 = dummy;
        //找到倒数第n+1个节点
        for(int i = 0; i <= n; i++){
            p1 = p1->next;
        }
        while(p1 != nullptr){
            p1 = p1->next;
            p2 = p2->next;
        }
        p2->next = p2->next->next;
        return dummy->next;
    }
};

141.环形链表

// 简单
快慢指针

class Solution {
public:
    bool hasCycle(ListNode *head) {
        ListNode* fast = head;
        ListNode* slow = head;
        while(fast && fast->next){
            fast = fast->next->next;
            slow = slow->next;
            if(slow == fast) return true;
        }
        return false;
    }
};

142. 环形链表 II

// 中等
快慢指针

class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        ListNode* fast = head;
        ListNode* slow = head;
        while (fast != NULL && fast->next != NULL){
            fast = fast->next->next;
            slow = slow->next;
            if(fast == slow) break;
        }
        if (fast == NULL || fast->next == NULL) return NULL;
        slow = head;
        while(slow != fast){
            fast =  fast->next;
            slow = slow->next;
        }
        return slow;
    }
};

160. 相交链表

// 简单
双指针

class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        ListNode* p1 = headA;
        ListNode* p2 = headB;
        while (p1 != p2){
            if(p1 == NULL) p1 = headB;
            else p1 = p1->next;
            if(p2 == NULL) p2 = headA;
            else p2 = p2->next;
        }
        return p1;
    }
};

        if(head == nullptr || head->next == nullptr) return head;
        ListNode* last = reverseList(head->next);
        head->next->next = head;
        head->next = nullptr;
        return last;
    }
};

92. 反转链表 II

// 中等

递归
反转链表的一部分 m~n，拆分为两个问题：
1. base case m=1，反转链表的前n个节点（head~n）：需要记录第n+1个节点successor，并且反转完之后连接，head->next = successor
2. 通常 m!=1，递归反转剩下的部分head->next，触发base case

class Solution {
public:
    ListNode* reverseBetween(ListNode* head, int left, int right) {
        if(left == 1) return reverseN(head, right);
        head->next = reverseBetween(head->next, left-1, right-1);
        return head;
    }
    ListNode* succesor = nullptr;
    ListNode* reverseN(ListNode* head, int n){
        if(n == 1){
            succesor = head->next;
            return head;
        }
        ListNode* last = reverseN(head->next, n-1);
        head->next->next = head;
        head->next = succesor;
        return last;
    }
};

25. K 个一组翻转链表

//困难
递归

class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        ListNode* cur_start = head;
        ListNode* next_start = head;
        for(int i = 0; i<k; i++){
            if (next_start == nullptr) return head; //base case：不足k个直接返回
            next_start = next_start->next; //先走k步，到下一组的起点
        }
        ListNode* newhead = reverse(cur_start, next_start); //反转k个的一组
        cur_start->next = reverseKGroup(next_start, k); //连接这一组的尾节点和下一组的头节点
        return newhead;
    }
    //迭代反转部分链表
    ListNode* reverse(ListNode* a, ListNode* b){
        ListNode* pre = nullptr;
        ListNode* cur = a;
        ListNode* nxt = a;
        while(cur != b){ // cur != nullptr 则为迭代反转整个链表
            nxt = cur->next;
            cur->next = pre;
            pre = cur;
            cur = nxt;
        }
        return pre;
    }
};

234. 回文链表

// 简单

  1. 快慢指针, 反转后半段链表:
    1. 找到链表的中间节点，一分为二
    2. 若总数是奇数，则慢指针最后需再走一步(判断条件到：fast->next!=NULL)
      (若判断条件到fast->next->next!=NULL, 则无需多走，但要加base case: head==NULL)
  2. 双指针, 逐个比较两个链表的节点值是否相等
  3. 恢复链表
  4. 返回结果
    时间复杂度O(n)，空间复杂度O(1)
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        if(head == nullptr) return true;

        ListNode* slow = head;
        ListNode* fast = head;
        // 找到链表中间节点
        while(fast->next != nullptr && fast->next->next != nullptr){
            slow = slow->next;
            fast = fast->next->next;
        }
        // 记录前半段的尾节点，便于恢复链表
        ListNode* endofhalf = slow;
        // 反转后半段链表
        ListNode* left = head;
        ListNode* right = reverse(slow->next);
        // 记录后半段的头节点，便于恢复链表
        ListNode* startofnexthalf = right;
        //比较回文
        while(right != nullptr){
            if(left->val != right->val) return false;
            left = left->next;
            right = right->next;
        }
        //恢复链表
        endofhalf->next = reverse(startofnexthalf);
        //返回结果
        return true;
    }
    ListNode* reverse(ListNode* head){
        if(head == nullptr) return nullptr;
        ListNode* pre = nullptr;
        ListNode* cur = head;
        ListNode* next = head;
        while(cur != nullptr){
            next = cur->next;
            cur->next = pre;
            pre = cur;
            cur = next;
        }
        return pre;
    }
};
