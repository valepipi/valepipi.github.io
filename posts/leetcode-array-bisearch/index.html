二分查找：搜索左边界、搜索右边界

  • 二分查找框架
//搜索左边界的二分查找
int left_bound(vector<int>& nums, int target){
    int left = 0, right = nums.size()-1;
    while(left <= right){
        int mid = left + (right - left)/2;
        if(nums[mid] < target){
            left = mid + 1;
        }else{
            right = mid - 1;
        }
    }
    if(left >= nums.size() || nums[left] != target) return -1;
    return left;
}

//搜索右边界的二分查找
int right_bound(vector<int>& nums, int target){
    int left = 0, right = nums.size()-1;
    while(left <= right){
        int mid = left + (right - left)/2;
        if(nums[mid] > target){
            right = mid - 1;
        }else{
            left = mid + 1;
        }
    }
    if(right < 0 || nums[right] != target) return -1;
    return right;
}

int binary_search(vector<int>& nums, int target){
    int left = 0, right = nums.size()-1;
    for(left <= right){
        int mid = left + (right - left)/2;
        if(nums[mid] < target){

        }else if(nums[mid] > target){
            right = mid - 1;
        }else if(nums[mid] == target){

二分查找的变体

528. 按权重随机选择

// 中等
思路： - 权重数组转化为前缀和 PreSum - 生成一个取值在PreSum之内的随机数 target - target右边最近的元素 -> 在数组PreSum中寻找大于等于target的最小元素: 搜索左侧边界的二分搜索

  1. 权重数组转为前缀和PreSum
vector<int> w; // 权重数组
vector<int> preSum; // 前缀和
int n = w.size();
preSum.resize(n + 1); // preSum[0] = 0 是占位符
preSum[0] = 0;
for(int i = 0; i < n; i++)
    preSum[i+1] = preSum[i] + w[i]
  1. 生成随机数 target in [1,preSum[n-1]]
int n = sizeof(preSum) / sizeof(preSum[0])
int target = rand() % preSum[n-1] + 1;
  1. 搜索左侧边界的二分搜索
int left_bound(vector<int>& nums, int target){
    if(nums.size() == 0) return -1;
    int left = 0, right = nums.size()-1;
    while(left <= right){
        int mid = left + (right - left)/2;
        if(nums[mid] < target)
            left = mid + 1;
        else
            right = mid - 1;
    }
    // 允许 mid[left] != target，本题目标是搜索 >= target的最小元素的索引
    if (left >= nums.size()) return -1;
    return left;
}

题解

class Solution {
private:
    vector<int> preSum;
public:
    Solution(vector<int>& w) {
        int n = w.size();
        preSum.resize(n+1);
        preSum[0] = 0;
        for(int i = 0; i < n; i++)
            preSum[i+1] = preSum[i] + w[i];
    }

    int pickIndex() {
        int n = preSum.size();
        int target = rand() % preSum[n-1] + 1; // [1, preSum[n-1]]
        return left_bound(preSum, target) - 1; // preSum[0]未利用
    }

    int left_bound(vector<int>& nums, int target){
        int left = 0, right = nums.size() - 1;
        while(left <= right){
            int mid = left + (right - left) / 2;
            if(nums[mid] < target)
                left = mid + 1;
            else
                right = mid - 1;
        }
        return left;
    }
};

410. 分割数组的最大值 //困难

反向思考，问题转化为二分查找： - 原问题：把数组nums分割为m个子数组subnums，找出其中一种划分方式 - max(getSum(subnum))最小 - 二分查找问题：把数组nums分割为一些子数组subnums，使其最大的数组和不超过max - max(getSum(subnum)) <= max ，子数组的个数至少是多少？当个数==m的时候满足题意, 返回最小的max

int splitArray(vector<int>& nums, int m){
    // 子数组和的范围：[最大元素值，所有元素和]
    int lo = getMax(nums), hi = getSum(nums) + 1;
    while(lo < hi){
        int mid = lo + (hi - lo)/2;
        int n = split(nums, mid);
        if(n == m){
            hi = mid; // 寻找左边界：在[lo, hi]中搜索·最小的max·
        }else if(n < m){ // 分成4个就能满足split，但要求分成五个，说明条件太宽松了
            hi = mid; // 收缩条件，压低上限
        }else if(n > m){
            lo = mid + 1;
        }
    }
    return lo;
}

int split(vector<int>& nums, int max){
    int count = 1;
    int sum = 0;
    for(int i = 0; i < nums.size(); i++){
        if (sum + nums[i] > max){
            count++;
            sum = nums[i];
        }else{
            sum += nums[i];
        }
    }
    return count;
}

int getMax(vector<int>& nums){
    int res = 0;
    for(int n : nums)
        res = max(n, res);
    return res;
}

int getSum(vector<int>& nums){
    int res = 0;
    for(int n: nums)
        res += n;
    return res;
}
