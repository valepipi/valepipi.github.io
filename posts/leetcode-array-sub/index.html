差分数组: 主要适用场景是频繁对原始数组的某个区间的元素进行增减。

  • 差分数组工具类
class Difference{
public:
    vector<int> diff;
    Difference(vector<int>& nums){
        assert(nums.size() > 0);
        diff = vector<int>(nums.size());
        diff[0] = nums[0];
        for(int i = 1; i<nums.size(); i++){
            diff[i] = nums[i] - nums[i-1];
        }
    }
    void increment(int i, int j, int val){
        diff[i] += val;
        if(j+1 < diff.size()) // 注意
            diff[j+1] -= val;
    }
    vector<int> result(){
        vector<int> res(diff.size());
        res[0] = diff[0];
        for(int i = 1; i< diff.size(); i++)
            res[i] = res[i-1] + diff[i];
        return res;
    }
};

307. 区间加法

// 中等

差分数组: 主要适用场景是频繁对原始数组的某个区间的元素进行增减。

  1. 构造差分数组：diff[i] = nums[i] - nums[i - 1]
  2. 对区间 nums[i..j] 的元素全部加 3，那么只需要让 diff[i] += 3，然后再让 diff[j+1] -= 3 即可
  3. 根据diff可以反推出原数组

vector<int> getModifiedArray(int length, vector<vector<int>& updates>){
    vector<int> nums(length, 0);
    Difference df(nums, length);
    for(auto: updates){
        int i = update[0];
        int j = update[1];
        int val = update[2];
        df.increment(i, j, val);
    }
    return df.result();

class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        vector<int> nums(n);
        Difference df(nums);
        for(auto& booking: bookings){
            int i = booking[0] - 1;
            int j = booking[1] - 1;
            int val = booking[2];
            df.increment(i, j, val);
        }
        return df.result();
    }
private:
    class Difference{
    public:

            res[0] = diff[0];

1094. 拼车

// 中等
差分数组，操作区间

class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        vector<int> nums(1001); //车站数量不超过1000
        Difference df(nums, 1001);
        for(auto& trip: trips){
            int i = trip[1];
            int j = trip[2] - 1; //j站已经下车
            int val = trip[0];
            df.increment(i, j, val);
            if (!df.cover(capacity)) return false;
        }
        return true;
    }
private:
    class Difference{
    public:
        vector<int> diff;
        Difference(vector<int>& nums, int len){
            assert(len > 0);
            diff = vector<int>(len);
            diff[0] = nums[0];
            for(int i = 1; i<len; i++)
                diff[i] = nums[i] - nums[i-1];
        }
        void increment(int i, int j, int val){
            diff[i] += val;
            if(j+1 < diff.size())
                diff[j+1] -= val;
        }
        bool cover(int max){
            vector<int> res(diff.size());
            res[0] = diff[0];
            if(res[0] > max) return false;
            for(int i = 1; i<diff.size(); i++){
                res[i] = res[i-1] + diff[i];
                if(res[i] > max) return false;
            }
            return true;
        }
    };
};

超载判断条件：

  • 问题：每次操作之后判断所有车站
    • df.cover(capacity)在循环中调用，开销是O(N)
  • 优化：所有操作之后判断所有车站
    • 由于车在每个时间点只能出现在一个站，所以对于一个站只可能操作一次，不可能有前后两批乘客在同一个站上车
    • 只需要对最后的结果(走完1000个车站)做一次遍历，判断每个站是否超载即可

        vector<int> res = df.result();
        for(auto& r: res){
