class Solution {

    void rotate(vector<vector<int>>& matrix) {
        // æ²¿ç€ä¸»å¯¹è§’çº¿é•œåƒç¿»è½¬
        int n = matrix.size();
        for(int i = 0; i<n; i++){
            for(int j =i; j<n; j++){
                int temp = matrix[i][j];
                matrix[i][j] = matrix[j][i];
                matrix[j][i] = temp;
            }
        }
        // ç¿»è½¬æ¯è¡Œçš„ä¸€ç»´æ•°ç»„
        for(auto& row: matrix){
            int i = 0;
            int j = row.size() - 1;
            while(i<j){
                int temp = row[j];
                row[j] = row[i];
                row[i] = temp;
                i++;
                j--;
            }
        }
    }
};

class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        int m = matrix.size();
        int n = matrix[0].size();
        vector<int> res;
        int upper_bound = 0;
        int lower_bound = m - 1;
        int left_bound = 0;
        int right_bound = n - 1;
        while(res.size() < m*n){
            // å·¦åˆ°å³
            if(upper_bound <= lower_bound){
                for(int j = left_bound; j <= right_bound; j++)
                    res.push_back(matrix[upper_bound][j]);
                // ä¸Šè¾¹ç•Œæ”¶ç¼©
                upper_bound++;
            }
            // ä¸Šåˆ°ä¸‹
            if(left_bound <= right_bound){
                for(int i = upper_bound; i <= lower_bound; i++)
                    res.push_back(matrix[i][right_bound]);
                // å³è¾¹ç•Œæ”¶ç¼©
                right_bound--;
            }
            // å³åˆ°å·¦
            if(upper_bound <= lower_bound){
                for(int j = right_bound; j >= left_bound; j--)
                    res.push_back(matrix[lower_bound][j]);
                lower_bound--;
            }
            // ä¸‹åˆ°ä¸Š

                for(int i = lower_bound; i >= upper_bound; i--)
                    res.push_back(matrix[i][left_bound]);
                left_bound++;
            }
        }
        return res;
    }
};
