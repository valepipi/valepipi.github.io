class PrefixSum{
private:
    int* prefix;
public:
    PrefixSum(vector<int>& nums){
        prefix = new int[nums.size()+1];
        for(int i = 0; i < nums.size(); i++)
            prefix[i+1] = prefix[i] + nums[i];
    }
    int query(int i, int j){
        return prefix[j+1] - prefix[i];
    }
}

差分数组工具类 

class Difference{
public:
    vector<int> diff;
    Difference(vector<int>& nums){
        assert(nums.size() > 0);
        diff = vector<int>(nums.size());
        diff[0] = nums[0];
        for(int i = 1; i<nums.size(); i++){
            diff[i] = nums[i] - nums[i-1];
        }
    }
    void increment(int i, int j, int val){
        diff[i] += val;
        if(j+1 < diff.size()) // 注意
            diff[j+1] -= val;
    }
    vector<int> result(){
        vector<int> res(diff.size());
        res[0] = diff[0];
        for(int i = 1; i< diff.size(); i++)
            res[i] = res[i-1] + diff[i];
        return res;
    }
};

26. 删除有序数组中的重复项

//简单
原地修改，双指针

  1. slow记录当前元素，fast在前方试探下一个不同的元素
  2. fast遇到不同的元素时，赋值给slow的下一个位置 (数组有序)
  3. 当fast遍历完数组，结束；此时，0-slow是去重后的数组
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int slow = 0;
        int fast = 0;
        if(nums.size() == 0) return 0;
        while(fast < nums.size()){
            if(nums[slow] != nums[fast]){
                slow++;
                nums[slow] = nums[fast];
            }
            fast++;
        }
        return slow+1;
    }
};

83. 删除排序链表中的重复元素

// 简单
26. 删除有序数组中的重复项的变体
原地修改，双指针
数组变为链表，下标操作变为指针操作

class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if(head == nullptr) return nullptr;
        ListNode* slow = head;
        ListNode* fast = head;
        while(fast != nullptr){
            if(slow->val != fast->val){
                slow->next = fast; //慢指针连接新节点，中间的重复元素都丢弃
                slow = slow->next; //慢指针向后移动一位
            }
            fast = fast->next;
        }
        slow->next = nullptr; //最后断开末尾的重复结点
        return head;
    }
};

class Solution {
public:
    void reverseString(vector<char>& s) {
        int slow = 0;
        int fast = s.size()-1;
        while(slow < fast){
            char tmp = s[slow];
            s[slow] = s[fast];
            s[fast] = tmp;
            slow++;
            fast--;
        }
    }
};

class Solution {
public:
    vector<int> twoSum(vector<int>& numbers, int target) {
        int slow = 0;
        int fast = numbers.size()-1;
        while(slow != fast){
            if(numbers[slow] + numbers[fast] == target) return {slow+1, fast+1};
            if(numbers[slow] + numbers[fast] < target) slow++;
            if(numbers[slow] + numbers[fast] > target) fast--;
        }
        return {-1, -1};
    }
};

5. 最长回文子串

// 中等
双指针，回文串分奇偶

  1. 从中间向两边找回文串，中心点分奇、偶：
    1. 奇数，以i为中心的最长回文串
    2. 偶数，以i, i+1为中心的最长回文串
  2. 选出较长的赋值给res
class Solution {
public:
    string longestPalindrome(string s) {
        string res = "";
        for(int i =0; i < s.length(); i++){
            string s1 = SubPalindrome(s, i, i);
            string s2 = SubPalindrome(s, i, i+1);
            if(s1.length()>res.length()) res = s1;
            if(s2.length()>res.length()) res = s2;
        }
        return res;
    }
    string SubPalindrome(string s, int slow, int fast){
        while(slow >= 0 && fast <= s.length()-1 && s[slow] == s[fast]){
            slow--;
            fast++;
        }
        return s.substr(slow+1, fast-slow-1);
    }
};

303. 区域和检索 - 数组不可变

// 简单

  • 暴力法：每次遍历数组求和，时间O(N)
  • 前缀和：时间O(1)

前缀和：主要适用的场景是原始数组不会被修改的情况下，频繁查询某个区间的累加和。

  1. 构造前缀和 PreSum[i] = nums[0]~nums[i](tips:PreSum[0]=0，空出)
  2. 查询 sumRange[left, right] = PreSum[right+1] - PreSum[left]

class NumArray {
private:
    vector<int> PreSum;
public:
    NumArray(vector<int>& nums) {
        PreSum.resize(nums.size()+1); // 默认初始化为0
        for(int i = 0; i<nums.size(); i++){
            PreSum[i+1] = PreSum[i] + nums[i];
        }
    }
    int sumRange(int left, int right) {
        return PreSum[right+1] - PreSum[left];
    }
};

304. 二维区域和检索 - 矩阵不可变

// 中等

构造前缀和, 利用已知的子问题的结果，通过计算简化检索的复杂度(空间换时间)
构造前缀和注意多申请一个空间

  • 初始化一维数组
vector<int> PreSum; // vector动态分配空间，初始没有大小
PreSum.resize(nums.size()+1); // 默认初始化为0
  • 初始化二维数组
vector<vector<int>> PreSum;
vector<vector<int>> PreSum(m+1, vector<int>(n+1, 0-));


class NumMatrix {
private:
    vector<vector<int>> PreSum;
public:
    NumMatrix(vector<vector<int>>& matrix) {
        int m = matrix.size();
        int n = matrix[0].size();
        if(m == 0 || n == 0) return;
        PreSum = vector<vector<int>> (m+1, vector<int>(n+1, 0));
        for(int i = 0; i<m; i++)
            for(int j = 0; j<n; j++)
                PreSum[i+1][j+1] = PreSum[i][j+1] + PreSum[i+1][j] - PreSum[i][j] + matrix[i][j];
    }

    int sumRegion(int row1, int col1, int row2, int col2) {
        return PreSum[row2+1][col2+1] - PreSum[row2+1][col1] - PreSum[row1][col2+1] + PreSum[row1][col1];
    }
};

1094. 拼车

// 中等
差分数组，操作区间

class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        vector<int> nums(1001); //车站数量不超过1000
        Difference df(nums, 1001);
        for(auto& trip: trips){
            int i = trip[1];
            int j = trip[2] - 1; //j站已经下车
            int val = trip[0];
            df.increment(i, j, val);
            if (!df.cover(capacity)) return false;
        }
        return true;
    }
private:
    class Difference{
    public:
        vector<int> diff;
        Difference(vector<int>& nums, int len){
            assert(len > 0);
            diff = vector<int>(len);
            diff[0] = nums[0];
            for(int i = 1; i<len; i++)
                diff[i] = nums[i] - nums[i-1];
        }
        void increment(int i, int j, int val){
            diff[i] += val;
            if(j+1 < diff.size())
                diff[j+1] -= val;
        }
        bool cover(int max){
            vector<int> res(diff.size());
            res[0] = diff[0];
            if(res[0] > max) return false;
            for(int i = 1; i<diff.size(); i++){
                res[i] = res[i-1] + diff[i];
                if(res[i] > max) return false;
            }
            return true;
        }
    };
};

超载判断条件：

  • 问题：每次操作之后判断所有车站
    • df.cover(capacity)在循环中调用，开销是O(N)

    • 由于车在每个时间点只能出现在一个站，所以对于一个站只可能操作一次，不可能有前后两批乘客在同一个站上车
    • 只需要对最后的结果(走完1000个车站)做一次遍历，判断每个站是否超载即可

48. 旋转图像

// 中等

二维矩阵顺时针旋转 - 沿着主对角线翻转（转置） - 翻转每行的一维数组

class Solution {
public:
    void rotate(vector<vector<int>>& matrix) {
        // 沿着主对角线镜像翻转
        int n = matrix.size();
        for(int i = 0; i<n; i++){
            for(int j =i; j<n; j++){
                int temp = matrix[i][j];
                matrix[i][j] = matrix[j][i];
                matrix[j][i] = temp;
            }
        }
        // 翻转每行的一维数组
        for(auto& row: matrix){
            int i = 0;
            int j = row.size() - 1;
            while(i<j){
                int temp = row[j];
                row[j] = row[i];
                row[i] = temp;
                i++;
                j--;
            }
        }
    }
};

54. 螺旋矩阵

// 中等

  • 螺旋遍历顺序：左到右，上到下，右到左，下到上
  • 定义四个边界
    • upper_bound = 0
    • lower_bound = m - 1
    • left_bound = 0
    • right_bound = n - 1
  • 朝四个方向遍历的条件与范围
    • 左到右
      • 条件：upper_bound <= lower_bound
      • 范围：left_bound ~ right_bound
    • 上到下
      • 条件：left_bound <= right_bound
      • 范围：upper_bound ~ lower_bound
    • 右到左
      • 条件：upper_bound <= lower_bound
      • 范围：right_bound ~ left_bound
    • 下到上
      • 条件：left_bound <= right_bound
      • 范围：lower_bound ~ uppper_bound
  • 螺旋收缩
    • 左到右：每轮上边界收缩 upper_bound++
    • 上到下：每轮右边界收缩 right_bound--
    • 右到左：每轮下边界收缩 lower_bound--
    • 下到上：每轮左边界收缩 left_bound++

class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        int m = matrix.size();
        int n = matrix[0].size();
        vector<int> res;
        int upper_bound = 0;
        int lower_bound = m - 1;
        int left_bound = 0;
        int right_bound = n - 1;
        while(res.size() < m*n){
            // 左到右
            if(upper_bound <= lower_bound){
                for(int j = left_bound; j <= right_bound; j++)
                    res.push_back(matrix[upper_bound][j]);
                // 上边界收缩
                upper_bound++;
            }
            // 上到下
            if(left_bound <= right_bound){
                for(int i = upper_bound; i <= lower_bound; i++)
                    res.push_back(matrix[i][right_bound]);
                // 右边界收缩
                right_bound--;
            }
            // 右到左
            if(upper_bound <= lower_bound){
                for(int j = right_bound; j >= left_bound; j--)
                    res.push_back(matrix[lower_bound][j]);
                lower_bound--;
            }
            // 下到上
            if(left_bound <= right_bound){
                for(int i = lower_bound; i >= upper_bound; i--)
                    res.push_back(matrix[i][left_bound]);
                left_bound++;
            }
        }
        return res;
    }
};

class Solution {
public:
    vector<vector<int>> generateMatrix(int n) {
        int num = 1;
        vector<vector<int>> matrix (n, vector<int>(n, 0));
        int upper_bound = 0;
        int lower_bound = n - 1;
        int left_bound = 0;
        int right_bound = n - 1;
        while(num <= n*n){
            // 左到右
            if(upper_bound <= lower_bound){
                for(int j = left_bound; j <= right_bound; j++)
                    matrix[upper_bound][j] = num++;
                // 上边界收缩
                upper_bound++;
            }
            // 上到下
            if(left_bound <= right_bound){
                for(int i = upper_bound; i <= lower_bound; i++)
                    matrix[i][right_bound] = num++;
                // 右边界收缩
                right_bound--;
            }
            // 右到左
            if(upper_bound <= lower_bound){
                for(int j = right_bound; j >= left_bound; j--)
                    matrix[lower_bound][j] = num++;
                lower_bound--;
            }
            // 下到上
            if(left_bound <= right_bound){
                for(int i = lower_bound; i >= upper_bound; i--)
                    matrix[i][left_bound] = num++;
                left_bound++;
            }
        }
        return matrix;
    }

76. 最小覆盖子串

public:
    string minWindow(string s, string t) {
        unordered_map<char, int> window, need;
        int left = 0, right = 0;
        int start = 0, min_len = INT_MAX;
        int valid = 0;
        for(char c: t) need[c]++;
        while(right < s.size()){
            char c = s[right];
            right++;
            if(need.count(c)){ //need.count(key) 查询key是否存在, 存在则返回1
                window[c]++;
                if(need[c] == window[c]) valid++;
            }
            while(valid == need.size()){
                int cur_len = right - left;
                if(cur_len < min_len){
                    start = left;
                    min_len = cur_len;
                }
                char d = s[left];
                left++;
                if(need.count(d)){
                    if(need[d] == window[d]) valid--;
                    window[d]--;
                }
            }
        }
        return min_len == INT_MAX?
            "": s.substr(start, min_len);
    }
};

3. 无重复字符的最长子串

// 中等
滑动窗口框架
76. 最小覆盖子串的变体
收缩条件：重复 while(window[c]>1)

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int left = 0, right = 0;
        int res = 0;

        unordered_map<char, int> window;

        while(right < s.size()){
            char c = s[right];
            right++;
            window[c]++;
            while(window[c]>1){
                char d = s[left];
                left++;
                window[d]--;
            }
            res = max(res, right-left);
        }
        return res;
    }
};

567. 字符串的排列

// 中等 滑动窗口框架
76. 最小覆盖子串的变体

  • 收缩条件：满足子串长度 while(right - left >= s1.size())
  • 返回条件：所有字符都满足数量 if(valid == need.size()) return true;
class Solution {
public:
    bool checkInclusion(string s1, string s2) {
        int left = 0, right = 0;
        int valid = 0;
        unordered_map<char, int> window, need;
        for(char c: s1) need[c]++;
        while(right < s2.size()){
            char c = s2[right];
            right++;
            if(need.count(c)){
                window[c]++;
                if(window[c] == need[c]) valid++;
            }
            while(right - left >= s1.size()){
                if(valid == need.size()) return true;
                char d = s2[left];
                left++;
                if(need.count(d)){
                    if(window[d] == need[d]) valid--;
                    window[d]--;
                }
            }
        }
        return false;
    }
};

438. 找到字符串中所有字母异位词

// 中等
滑动窗口框架

异位词： 排列
即：找出所有子串的排列

  • 收缩条件：while(right - right >= p.size())
  • 返回结果：if(valid == need.size()) res.push_back(left);
class Solution {
public:
    vector<int> findAnagrams(string s, string p) {
        if (p.size() > s.size()) return {};
        vector<int> res;
        int left = 0, right = 0;
        int valid = 0;
        unordered_map<char, int> window, need;
        for(char c: p) need[c]++;
        while(right < s.size()){
            char c = s[right];
            right++;
            if(need.count(c)){
                window[c]++;
                if(window[c] == need[c]) valid++;
            }
            while(right - left >= p.size()){
                if(valid == need.size()) res.push_back(left);
                char d = s[left];
                left++;
                if(need.count(d)){
                    if(window[d] == need[d]) valid--;
                    window[d]--;
                }
            }
        }
        return res;
    }
};

187. 重复的DNA序列

// 中等

法一：暴力搜索
时间复杂度：O(NL)

class Solution {
public:
    vector<string> findRepeatedDnaSequences(string s) {
        int n = s.length();
        unordered_set<string> seen;
        unordered_set<string> res;
        for(int i = 0; i+10 <= n; i++){ // O(N)
            string subStr = s.substr(i, 10); // O(L) 待优化
            if(seen.count(subStr))
                res.insert(subStr);
            else
                seen.insert(subStr);
        }
        return vector<string> (res.begin(), res.end());
    }
};

法二：滑动哈希
时间复杂度：O(N)

滑动哈希代替暴力匹配：不每次都去一个字符一个字符地比较子串和模式串，而是维护一个滑动窗口，运用滑动哈希算法一边滑动一边计算窗口中字符串的哈希值，拿这个哈希值去和模式串的哈希值比较，这样就可以避免截取子串，从而把匹配算法降低为 O(N)

优化：

  • 用唯一的数字(哈希思想)代表对应的字符串，减少字符串的构造时间 O(L) -> O(1)
    • 在遍历中不断构造字符串： 每次都需要 s.subtr(pos, L) ，时间O(L)
    • 在遍历中不断构造数字：四个字母 AGCT -> 四进制元 0123；长度L=10的碱基序列 -> 10位的四进制数字；每次只需删除最高位，添加最低位就能构造下一个数字，时间 O(1)
  • 用 int 即可囊括 10 位四进制数字的最大值，免于用long，节省空间
    • 10位的四进制数字包含的哈希值个数(几种可能性)：4^10 = 1,048,576
    • int 范围 4Bytes 包含的哈希值个数(几种可能性)： 2^32 = 4,294,967,296, 其中 2,4,294,967,296是十位数，足够表示10位的四进制数字 2^32 > 4^10（全部转为十进制）
数据范围的另一种算法（全部转为二进制）：  
int类型是32位二进制  
十位四进制数最大值：3,333,333,333  
-> 二进制数：11,111,111,111,111,111,111 (20位<=32位，可以容纳)  
十位十进制数最大值：9,999,999,999  
-> 二进制数：1,001,010,100,000,010,111,110,001,111,111,111 (34位>32位，不能容纳)  


class Solution {
public:
    vector<string> findRepeatedDnaSequences(string s) {
        vector<int> nums(s.length());
        // 四个字母转换为四进制的四个数字，之后用int表示10个字母的序列
        for(int i = 0; i < nums.size(); i++){
            switch(s[i]){
                case 'A':
                    nums[i] = 0;
                    break;
                case 'G':
                    nums[i] = 1;
                    break;
                case 'T':
                    nums[i] = 2;
                    break;
                case 'C':
                    nums[i] = 3;
                    break;
            }
        }
        unordered_set<int> seen; //记录见过的，用于判断是否重复出现
        unordered_set<string> res; // 重复出现的
        int L = 10; // 位数
        int R = 4; // 进制
        int RL = pow(R, L-1);
        int windowHash = 0;

        int left = 0, right = 0;
        while(right < nums.size()){
            windowHash = R * windowHash + nums[right]; // 向右滑动，低位添加
            right++;

            if(right - left == L){
                if(seen.count(windowHash)) // 重复了
                    res.insert(s.substr(left, right - left));
                else // 新的
                    seen.insert(windowHash);
                windowHash = windowHash - nums[left] * RL; // 左边收缩，高位删除
                left++;
            }
        }
    return vector<string> (res.begin(), res.end());
    }
};
